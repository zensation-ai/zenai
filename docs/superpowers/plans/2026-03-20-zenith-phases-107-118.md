// frontend/src/components/PlannerPage/usePlannerFilters.ts
import { useState, useMemo, useCallback } from 'react';
import {
  type PlannerFilters,
  type PlannerSort,
  type FilterChipDef,
  type TaskStatus,
  type TaskPriority,
  DEFAULT_FILTERS,
  DEFAULT_SORT,
} from './types';

type FilterGroup = 'status' | 'priority' | 'project' | 'special';

const STATUS_CHIPS: FilterChipDef[] = [
  { id: 'status-backlog', label: 'Backlog', group: 'status', value: 'backlog' },
  { id: 'status-todo', label: 'Offen', group: 'status', value: 'todo' },
  { id: 'status-in_progress', label: 'In Arbeit', group: 'status', value: 'in_progress' },
  { id: 'status-done', label: 'Erledigt', group: 'status', value: 'done' },
  { id: 'status-cancelled', label: 'Abgebrochen', group: 'status', value: 'cancelled' },
];

const PRIORITY_CHIPS: FilterChipDef[] = [
  { id: 'priority-high', label: 'Hoch', group: 'priority', value: 'high' },
  { id: 'priority-medium', label: 'Mittel', group: 'priority', value: 'medium' },
  { id: 'priority-low', label: 'Niedrig', group: 'priority', value: 'low' },
];

const SPECIAL_CHIPS: FilterChipDef[] = [
  { id: 'special-overdue', label: 'Überfällig', group: 'special', value: 'overdue' },
];

export function usePlannerFilters(initialTab?: string) {
  const [filters, setFilters] = useState<PlannerFilters>(() => ({
    ...DEFAULT_FILTERS,
    status: new Set(DEFAULT_FILTERS.status),
    priorities: new Set(DEFAULT_FILTERS.priorities),
    projects: new Set(DEFAULT_FILTERS.projects),
  }));
  const [sort, setSort] = useState<PlannerSort>(DEFAULT_SORT);

  const toggleFilter = useCallback((group: FilterGroup, value: string) => {
    setFilters(prev => {
      const next = { ...prev };
      if (group === 'status') {
        const set = new Set(prev.status);
        if (set.has(value as TaskStatus)) {
          set.delete(value as TaskStatus);
        } else {
          set.add(value as TaskStatus);
        }
        next.status = set;
      } else if (group === 'priority') {
        const set = new Set(prev.priorities);
        if (set.has(value as TaskPriority)) {
          set.delete(value as TaskPriority);
        } else {
          set.add(value as TaskPriority);
        }
        next.priorities = set;
      } else if (group === 'project') {
        const set = new Set(prev.projects);
        if (set.has(value)) {
          set.delete(value);
        } else {
          set.add(value);
        }
        next.projects = set;
      }
      return next;
    });
  }, []);

  const clearAll = useCallback(() => {
    setFilters({
      ...DEFAULT_FILTERS,
      status: new Set(DEFAULT_FILTERS.status),
      priorities: new Set<TaskPriority>(),
      projects: new Set<string>(),
    });
  }, []);

  const setSearch = useCallback((search: string) => {
    setFilters(prev => ({ ...prev, search }));
  }, []);

  const toggleOverdue = useCallback(() => {
    setFilters(prev => ({ ...prev, overdue: !prev.overdue }));
  }, []);

  const activeFilterCount = useMemo(() => {
    let count = 0;
    // Count non-default status selections
    filters.status.forEach(s => {
      if (!DEFAULT_FILTERS.status.has(s)) count++;
    });
    DEFAULT_FILTERS.status.forEach(s => {
      if (!filters.status.has(s)) count++;
    });
    count += filters.priorities.size;
    count += filters.projects.size;
    if (filters.overdue) count++;
    return count;
  }, [filters]);

  const chipDefs = useMemo<FilterChipDef[]>(
    () => [...STATUS_CHIPS, ...PRIORITY_CHIPS, ...SPECIAL_CHIPS],
    [],
  );

  return {
    filters,
    sort,
    setSort,
    toggleFilter,
    setSearch,
    clearAll,
    toggleOverdue,
    activeFilterCount,
    chipDefs,
  };
}

// frontend/src/components/PlannerPage/FilterChipBar.tsx
import { useMemo } from 'react';
import { X } from 'lucide-react';
import type { FilterChipDef, PlannerFilters } from './types';
import './FilterChipBar.css';

interface FilterChipBarProps {
  chips: FilterChipDef[];
  filters: PlannerFilters;
  onToggle: (group: string, value: string) => void;
  onClear: () => void;
  activeCount?: number;
}

function isChipActive(chip: FilterChipDef, filters: PlannerFilters): boolean {
  if (chip.group === 'status') return filters.status.has(chip.value as any);
  if (chip.group === 'priority') return filters.priorities.has(chip.value as any);
  if (chip.group === 'project') return filters.projects.has(chip.value);
  if (chip.group === 'special' && chip.value === 'overdue') return filters.overdue;
  return false;
}

export function FilterChipBar({ chips, filters, onToggle, onClear, activeCount = 0 }: FilterChipBarProps) {
  const grouped = useMemo(() => {
    const groups: { group: string; chips: FilterChipDef[] }[] = [];
    let currentGroup = '';
    for (const chip of chips) {
      if (chip.group !== currentGroup) {
        groups.push({ group: chip.group, chips: [] });
        currentGroup = chip.group;
      }
      groups[groups.length - 1].chips.push(chip);
    }
    return groups;
  }, [chips]);

  return (
    <div className="filter-chip-bar" role="toolbar" aria-label="Filter-Chips">
      <div className="filter-chip-bar__scroll">
        {grouped.map((g, gi) => (
          <div key={g.group} className="filter-chip-bar__group">
            {gi > 0 && <div className="filter-chip-bar__separator" />}
            {g.chips.map(chip => (
              <button
                key={chip.id}
                className={`filter-chip-bar__chip ${isChipActive(chip, filters) ? 'filter-chip-bar__chip--active' : ''}`}
                aria-pressed={isChipActive(chip, filters)}
                onClick={() => onToggle(chip.group, chip.value)}
              >
                {chip.label}
              </button>
            ))}
          </div>
        ))}
        {activeCount > 0 && (
          <button
            className="filter-chip-bar__clear"
            onClick={onClear}
            aria-label={`${activeCount} Filter entfernen`}
          >
            <X size={14} />
          </button>
        )}
      </div>
    </div>
  );
}

import { memo } from 'react';
import { CheckCircle2, Circle, Clock, AlertTriangle } from 'lucide-react';
import './TaskCard.css';

interface Task {
  id: string;
  title: string;
  description?: string;
  status: string;
  priority: string;
  due_date?: string | null;
  project_id?: string | null;
  project_name?: string | null;
}

interface TaskCardProps {
  task: Task;
  onClick: (task: Task) => void;
  view?: 'grid' | 'list';
}

const PRIORITY_LABELS: Record<string, string> = {
  high: 'Hoch',
  medium: 'Mittel',
  low: 'Niedrig',
};

const STATUS_ICONS: Record<string, typeof Circle> = {
  backlog: Circle,
  todo: Circle,
  in_progress: Clock,
  done: CheckCircle2,
  cancelled: AlertTriangle,
};

function formatDate(dateStr: string): string {
  const d = new Date(dateStr);
  return d.toLocaleDateString('de-DE', { day: '2-digit', month: '2-digit' });
}

export const TaskCard = memo(function TaskCard({ task, onClick, view = 'list' }: TaskCardProps) {
  const StatusIcon = STATUS_ICONS[task.status] ?? Circle;

  return (
    <article
      className={`task-card ${task.status === 'done' ? 'task-card--done' : ''}`}
      data-view={view}
      onClick={() => onClick(task)}
      role="button"
      tabIndex={0}
      onKeyDown={e => {
        if (e.key === 'Enter' || e.key === ' ') {
          e.preventDefault();
          onClick(task);
        }
      }}
    >
      <div className="task-card__status">
        <StatusIcon size={16} />
      </div>
      <div className="task-card__content">
        <h3 className="task-card__title">{task.title}</h3>
        {view === 'grid' && task.description && (
          <p className="task-card__description">{task.description}</p>
        )}
      </div>
      <div className="task-card__meta">
        <span className={`task-card__priority task-card__priority--${task.priority}`}>
          {PRIORITY_LABELS[task.priority] ?? task.priority}
        </span>
        {task.due_date && (
          <span className="task-card__due">{formatDate(task.due_date)}</span>
        )}
        {task.project_name && (
          <span className="task-card__project">{task.project_name}</span>
        )}
      </div>
    </article>
  );
});

// frontend/src/components/PlannerPage/TaskListView.tsx
import { useRef } from 'react';
import { useVirtualizer } from '@tanstack/react-virtual';
import { TaskCard } from './TaskCard';

interface Task {
  id: string;
  title: string;
  description?: string;
  status: string;
  priority: string;
  due_date?: string | null;
  project_id?: string | null;
  project_name?: string | null;
}

interface TaskListViewProps {
  tasks: Task[];
  onTaskClick: (task: Task) => void;
}

const ROW_HEIGHT = 56;
const GAP = 4;

export function TaskListView({ tasks, onTaskClick }: TaskListViewProps) {
  const parentRef = useRef<HTMLDivElement>(null);

  const virtualizer = useVirtualizer({
    count: tasks.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => ROW_HEIGHT + GAP,
    overscan: 5,
  });

  if (tasks.length === 0) return null;

  const virtualItems = virtualizer.getVirtualItems();
  const itemsToRender = virtualItems.length > 0
    ? virtualItems
    : tasks.map((_, i) => ({ key: i, index: i, start: i * (ROW_HEIGHT + GAP), size: ROW_HEIGHT + GAP }));

  return (
    <div ref={parentRef} className="task-list-view" style={{ overflow: 'auto', flex: 1 }}>
      <div style={{ height: virtualizer.getTotalSize() || tasks.length * (ROW_HEIGHT + GAP), position: 'relative' }}>
        {itemsToRender.map(row => {
          const task = tasks[row.index];
          if (!task) return null;
          return (
            <div
              key={row.key}
              style={{
                position: 'absolute',
                top: row.start,
                left: 0,
                right: 0,
                padding: '0 var(--space-4, 16px)',
              }}
            >
              <TaskCard task={task} onClick={onTaskClick} view="list" />
            </div>
          );
        })}
      </div>
    </div>
  );
}

import { Spinner } from '../../design-system/components/Spinner';
import { EmptyState } from '../../design-system/components/EmptyState';
import { QueryErrorState } from '../QueryErrorState';

import { TaskListView } from './TaskListView';
import { TaskCard } from './TaskCard';
import { PlannerPanel } from './PlannerPanel';
import { usePlannerFilters } from './usePlannerFilters';
import { useTasksQuery, useProjectsQuery } from '../../hooks/queries/useTasks';
import { useContactsQuery } from '../../hooks/queries/useContacts';
import type { PlannerViewMode, PlannerSmartPageProps } from './types';
import './PlannerSmartPage.css';

// Lazy-load heavy views
const CalendarPage = lazy(() => import('../CalendarPage').then(m => ({ default: m.default ?? m.CalendarPage ?? (() => null) })));
const KanbanBoard = lazy(() => import('../PlannerPage/KanbanBoard').then(m => ({ default: m.default ?? m.KanbanBoard ?? (() => null) })).catch(() => ({ default: () => <EmptyState title="Board nicht verfügbar" /> })));
const GanttChart = lazy(() => import('../PlannerPage/GanttChart').then(m => ({ default: m.default ?? m.GanttChart ?? (() => null) })).catch(() => ({ default: () => <EmptyState title="Timeline nicht verfügbar" /> })));

export function PlannerSmartPage({ context, initialTab }: PlannerSmartPageProps) {
  // 1. Filter state
  const { filters, sort, setSort, toggleFilter, setSearch, clearAll, toggleOverdue, activeFilterCount, chipDefs } =
    usePlannerFilters(initialTab);

  // 2. View mode + panel state
  const [viewMode, setViewMode] = useState<PlannerViewMode>('calendar');
  const [panelOpen, setPanelOpen] = useState(false);
  const [selectedItem, setSelectedItem] = useState<any | null>(null);
  const [selectedItemType, setSelectedItemType] = useState<'task' | 'event' | 'contact'>('task');

  // 3. React Query
  const { data: tasks = [], isLoading: tasksLoading, error: tasksError, refetch: refetchTasks } = useTasksQuery(context);
  const { data: projects = [] } = useProjectsQuery(context);
  const { data: contacts = [] } = useContactsQuery(context);

  // 4. Client-side filtering
  const filteredTasks = useMemo(() => {
    let result = tasks as any[];

    if (filters.status.size > 0) {
      result = result.filter(t => filters.status.has(t.status));
    }
    if (filters.priorities.size > 0) {
      result = result.filter(t => filters.priorities.has(t.priority));
    }
    if (filters.projects.size > 0) {
      result = result.filter(t => t.project_id && filters.projects.has(t.project_id));
    }
    if (filters.overdue) {
      const now = new Date().toISOString();
      result = result.filter(t => t.due_date && t.due_date < now && t.status !== 'done');
    }
    if (filters.search.trim()) {
      const q = filters.search.toLowerCase();
      result = result.filter(t =>
        t.title?.toLowerCase().includes(q) ||
        t.description?.toLowerCase().includes(q)
      );
    }

    // Sort
    return [...result].sort((a, b) => {
      const dir = sort.direction === 'asc' ? 1 : -1;
      if (sort.field === 'priority') {
        const order: Record<string, number> = { high: 0, medium: 1, low: 2 };
        return dir * ((order[a.priority] ?? 1) - (order[b.priority] ?? 1));
      }
      const va = (a[sort.field] ?? '') as string;
      const vb = (b[sort.field] ?? '') as string;
      return dir * va.localeCompare(vb);
    });
  }, [tasks, filters, sort]);

  // 5. Handlers
  const handleTaskClick = useCallback((task: any) => {
    setSelectedItem(task);
    setSelectedItemType('task');
    setPanelOpen(true);
  }, []);

  const handleNewTask = useCallback(() => {
    // Open panel with empty task for creation
    setSelectedItem(null);
    setSelectedItemType('task');
    setPanelOpen(true);
  }, []);

  // 6. Render
  if (tasksError) {
    return <QueryErrorState error={tasksError as Error} refetch={refetchTasks} />;
  }

  return (
    <div className="planner-smart-page">
      <FilterChipBar
        chips={chipDefs}
        filters={filters}
        onToggle={toggleFilter as (group: string, value: string) => void}
        onClear={clearAll}
        activeCount={activeFilterCount}
      />

      <PlannerToolbar
        viewMode={viewMode}
        onViewChange={setViewMode}
        search={filters.search}
        onSearchChange={setSearch}
        onNewTask={handleNewTask}
      />

      <div className="planner-smart-page__content">
        <Suspense fallback={<div className="planner-smart-page__loading"><Spinner size="md" /></div>}>
          {viewMode === 'calendar' && (
            <CalendarPage context={context} embedded={true} />
          )}
          {viewMode === 'board' && (
            <KanbanBoard
              tasks={filteredTasks}
              projects={projects}
              context={context}
            />
          )}
          {viewMode === 'list' && (
            filteredTasks.length === 0 && !tasksLoading ? (
              <EmptyState title="Keine Aufgaben gefunden" description="Erstelle eine neue Aufgabe oder ändere die Filter." />
            ) : (
              <TaskListView tasks={filteredTasks} onTaskClick={handleTaskClick} />
            )
          )}
          {viewMode === 'timeline' && (
            <GanttChart
              tasks={filteredTasks}
              projects={projects}
              context={context}
            />
          )}
          {viewMode === 'contacts' && (
            <div className="planner-smart-page__contacts">
              {(contacts as any[]).length === 0 ? (
                <EmptyState title="Keine Kontakte" description="Erstelle deinen ersten Kontakt." />
              ) : (
                <TaskListView
                  tasks={(contacts as any[]).map(c => ({
                    id: c.id,
                    title: c.name ?? `${c.first_name ?? ''} ${c.last_name ?? ''}`.trim(),
                    status: 'contact',
                    priority: 'medium',
                    description: c.email ?? c.organization_name ?? '',
                  }))}
                  onTaskClick={(item) => {
                    setSelectedItem(item);
                    setSelectedItemType('contact');
                    setPanelOpen(true);
                  }}
                />
              )}
            </div>
          )}
        </Suspense>
      </div>

      <PlannerPanel
        open={panelOpen}
        item={selectedItem}
        itemType={selectedItemType}
        onClose={() => setPanelOpen(false)}
        context={context}
      />
    </div>
  );
}

// frontend/src/hooks/queries/useEmails.ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { queryKeys } from '../../lib/query-keys';
import { apiClient } from '../../lib/api-client';

export function useEmailsQuery(context: string, folder = 'inbox', filters?: { search?: string; category?: string; unreadOnly?: boolean }) {
  return useQuery({
    queryKey: queryKeys.email.list(context, folder, filters),
    queryFn: async () => {
      const params = new URLSearchParams();
      if (folder !== 'inbox') params.set('status', folder);
      if (filters?.search) params.set('search', filters.search);
      if (filters?.category) params.set('category', filters.category);
      if (filters?.unreadOnly) params.set('status', 'unread');
      const res = await apiClient.get(`/api/${context}/emails?${params}`);
      return res.data?.data ?? res.data?.emails ?? [];
    },
